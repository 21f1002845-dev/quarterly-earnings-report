Risk-Adjusted Return Calculation (Code)


import pandas as pd

def risk_adjusted_return(df: pd.DataFrame) -> pd.Series:
    """
    Compute risk-adjusted return for each portfolio.
    df must contain columns: 'return', 'risk_free_rate', 'volatility'.
    """
    excess_return = df["return"] - df["risk_free_rate"]
    # Simple Sharpe-like ratio
    ratio = excess_return / df["volatility"].replace(0, float("nan"))
    return ratio
